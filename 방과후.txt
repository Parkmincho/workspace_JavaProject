import java.awt.Dimension;
import java.awt.Label;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.util.Random;

import javax.swing.JButton;
import javax.swing.JComboBox;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JPanel;
import javax.swing.JTextField;

public class GUITest11_UpDown {

	protected static int computer;

	public static void main(String[] args) {
		JFrame frame = new JFrame("업앤다운");
		JPanel panel = new JPanel();
		frame.setPreferredSize(new Dimension(400, 600));
		
		//JTextField, JButton
		JTextField tfInput = new JTextField(10);
		JButton btQusetion = new JButton("이냐?");
		//JLabel
		JLabel lbResult = new JLabel("1~100사이 숫자 말해봐");
		Random random = new Random();
		int answer = random.nextInt(100)+1; //1~100
		ActionListener al = new ActionListener() {
			
			@Override
			public void actionPerformed(ActionEvent e) {
				//사용자 입력값 가져오기
				String input = tfInput.getText();
				int player = Integer.parseInt(input);//String -> int
				//컴퓨터 값과 비교
				//비교
				if (computer > player) {
					lbResult.setText("UP");
				}
				
				else if (computer > player) {
					lbResult.setText("DOWN");
				}
				//결과
				else 
					//결과
					//탈출
					lbResult.setText("정답입니다.");
				
			}
		};
		
		panel.add(tfInput);
		panel.add(btQusetion);
		panel.add(lbResult);
		frame.add(panel);
		
		frame.pack();
		frame.setVisible(true);
		frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

	}

}
